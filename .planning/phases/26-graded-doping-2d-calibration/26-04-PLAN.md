       reset_devsim_fully()
       try:
           dev = create_dd_device(

                   f"got W={actual_W:.6e} cm, "
                   f"rel_diff = {abs(actual_W - expected_W) / expected_W:.3e}, tol = {tol}"
               )
       finally:
           reset_devsim_fully()
